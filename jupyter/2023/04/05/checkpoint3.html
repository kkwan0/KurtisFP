Built Sort into Data Structures 

import java.util.LinkedList;

public class QueueExample {
   public static void main(String[] args) {
      LinkedList<String> queue = new LinkedList<>();

      // Adding elements to the queue
      queue.add("John");
      queue.add("Sarah");
      queue.add("Michael");
      queue.add("David");
      System.out.println(queue);
      long startTime = System.currentTimeMillis();

      for (int i = 0; i < queue.size() - 1; i++) { //bubble sort
         for (int j = 0; j < queue.size() - i - 1; j++) {
             String current = queue.get(j);
             String next = queue.get(j+1);
             if (current.compareTo(next) > 0) {
                 queue.set(j, next);
                 queue.set(j+1, current);
             }
         }
     }
     long endTime = System.currentTimeMillis();
     long totalTime = endTime - startTime;
     System.out.println("Time taken to sort array: " + totalTime + " milliseconds");

     System.out.println("sorted queue: " + queue);
      // Printing and removing all elements in the queue
    }     
}

Big O analysis

Bubble Sort: usually o(n^2), can be o(n), only if it is already sorted

Insertion Sort: usually o(n^2), can be o(n), also if it is already sorted

Merge Sort: always o(n*log(n))

Selection Sort: always o(n^2)

    public static int[] bubblesort(Integer[] arr) {
        long startTime = System.currentTimeMillis();

        //start bubble sort
        int n = arr.length;
        int[] sortedArr = new int[n];
        for(int i = 0; i < n; i++) {
            sortedArr[i] = arr[i];
        }
        int nunm = 0;
        int compar =0;

        for (int i = 0; i < n - 1; i++) {
        
            for (int j = 0; j < n - i - 1; j++) {
                compar ++;
                if (sortedArr[j] > sortedArr[j+1]) {
                    // Swap elements
                    int temp = sortedArr[j];
                    sortedArr[j] = sortedArr[j+1];
                    sortedArr[j+1] = temp;
                    nunm++; //counts nmber of swaps
                }
            }
        }
        System.out.println("number of swaps for bubble: " + nunm);
        System.out.println("nnumber of comparisons in bubble sort: " + compar);
        long endTime = System.currentTimeMillis();
        long totalTime = endTime - startTime;
        System.out.println("Time taken to sort binary array: " + totalTime + " milliseconds");
        return sortedArr; // in bubblesort method

        //end of bubble sort
    }

    public static int[] selectionsort(Integer[] arr) {
        long startTime = System.currentTimeMillis();
        int compariso = 0;
        int swaps = 0;
        int n = arr.length;
        int[] sortedArr = new int[n];
    
        for (int i = 0; i < n; i++) {
            sortedArr[i] = arr[i].intValue(); // convert Integer to int
        }
    
        for (int i = 0; i < n - 1; i++) {
            int minIndex = i; //sets the minimum value
    
            for (int j = i + 1; j < n; j++) {
                if (sortedArr[j] < sortedArr[minIndex]) { //checks if the value is less than the original smallest value
                    minIndex = j;
                    compariso++;
                }
            }
    
            int temp = sortedArr[minIndex]; //makes the new minimum value into the temporary value
            sortedArr[minIndex] = sortedArr[i]; //makes the swap, the small value into the place that it is suppoed to be
            sortedArr[i] = temp; //
            swaps++;
        }
        System.out.println("comparisons in selection sort: " + compariso);
        System.out.println("swaps in selection sort: " + swaps);

        long endTime = System.currentTimeMillis();
        long totalTime = endTime - startTime;
        System.out.println("Time taken to sort selection array: " + totalTime + " milliseconds");
    
        return sortedArr;
    }
